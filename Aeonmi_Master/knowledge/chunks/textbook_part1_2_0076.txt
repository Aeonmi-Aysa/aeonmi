=== The Fidelity Measurement System ===

BODY:
The qiskit_runner.py implements a fidelity measurement system using Total Variation Distance (TVD). When you run the same circuit on both Aer and IBM Brisbane, fidelity measures how close the two result distributions are.

A fidelity of 1.0 means the real hardware produced exactly the Aer result. A fidelity of 0.0 means completely different distributions. In practice, well-calibrated IBM hardware on simple circuits achieves fidelity of 0.95–0.99.

The majority_vote function in qiskit_runner.py runs the circuit 3 times and takes the majority outcome — a simple form of noise mitigation.

CODE:
```
// Understanding fidelity in Aeonmi
// (This demonstrates the concept; actual fidelity requires two backend runs)

function compute_tvd(p, q, labels) {
    // Total Variation Distance between two distributions
    // TVD = 0.5 * sum |p(x) - q(x)|
    // Fidelity = 1 - TVD
    let tvd = 0.0;
    let i = 0;
    while i < len(labels) {
        let diff = p[i] - q[i];
        if diff < 0 { diff = 0 - diff; }
        tvd = tvd + diff;
        i = i + 1;
    }
    return 1.0 - (tvd / 2.0);
}

function main() {
    // Simulated comparison: Aer vs IBM Brisbane for Bell state (1024 shots)
    // Aer (ideal): 512 "00", 512 "11", 0 "01", 0 "10"
    // IBM Brisbane (with noise): 489 "00", 498 "11", 19 "01", 18 "10"

    let aer_counts    = [512.0, 512.0, 0.0, 0.0];    // 00, 11, 01, 10
    let ibm_counts    = [489.0, 498.0, 19.0, 18.0];
    let total = 1024.0;

    // Normalize to probabilities
    let aer_probs = [aer_counts[0]/total, aer_counts[1]/total,
                     aer_counts[2]/total, aer_counts[3]/total];
    let ibm_probs = [ibm_counts[0]/total, ibm_counts[1]/total,
                     ibm_counts[2]/total, ibm_counts[3]/total];

    let labels = ["00", "11", "01", "10"];
    let fidelity = compute_tvd(aer_probs, ibm_probs, labels);

    print("Aer distribution:    " + aer_probs);
    print("IBM Brisbane distribution: " + ibm_probs);
    print("Fidelity (1 - TVD):  " + fidelity);
    print("Interpretation: " + floor(fidelity * 100) + "% match between Aer and hardware");
}

main();
```

```
function abs_val(x) {
    if x < 0 { return 0 - x; }
    return x;
}

function fidelity_from_counts(a00, a11, b00, b11, shots) {
    // TVD between two Bell state distributions
    let p00 = a00 / shots;
    let p11 = a11 / shots;
    let q00 = b00 / shots;
    let q11 = b11 / shots;
    let tvd = (abs_val(p00 - q00) + abs_val(p11 - q11)) / 2.0;
    return 1.0 - tvd;
}

function main() {
    print("=== AER vs IBM BRISBANE COMPARISON ===");
    print("");
```

CODE:
```
// FORMAT: "n_q n_c shots op_count  [type target control] * op_count"
//
// FIELDS:
//   n_q      — number of qubits (int)
//   n_c      — number of classical bits (int)
//   shots    — measurement repetitions (int)
//   op_count — number of operations (int)
//
// OPERATION: three integers [type] [target] [control]
//   type:    0=H  1=X  2=Y  3=Z  4=CX/CNOT  5=S  6=T  7=MEASURE
//   target:  qubit index (0-based)
//   control: qubit index for two-qubit gates; -1 for single-qubit gates
//
// EXAMPLES:
// Single qubit H + measure:     "1 1 100 2   0 0 -1   7 0 0"
// X gate + measure:             "1 1 100 2   1 0 -1   7 0 0"
// Bell state (standard):        "2 2 512 4   0 0 -1   4 1 0   7 0 0   7 1 1"
// GHZ state (3 qubits):         "3 3 1024 6  0 0 -1   4 1 0   4 2 0   7 0 0   7 1 1   7 2 2"
// H + Z + measure:              "1 1 100 3   0 0 -1   3 0 -1   7 0 0"
```
