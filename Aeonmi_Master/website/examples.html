2. Shor's Algorithm — Integer Factorization

Speedup: Polynomial vs exponential classical

import quantum;

fn qft(qc, n) {
    ⍝ Quantum Fourier Transform
    let i = n - 1;
    while i >= 0 {
        qc.h(i);
        ⍝ controlled phase rotations...
        i = i - 1;
    }
    return qc;
}

    let qc = quantum.circuit(3);
    qc.h(0); qc.h(1); qc.h(2);
    qc = qft(qc, 3);
    ⍝ Extract period → factors

}
