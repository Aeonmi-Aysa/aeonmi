⍝ Create a Bell state (quantum entanglement)
function create_bell_state() {
    print("Creating Bell state |Φ+⟩ = (|00⟩ + |11⟩)/√2");
    
    ⍝ Initialize |00⟩
    let a00 = 1.0;
    let a01 = 0.0;
    let a10 = 0.0;
    let a11 = 0.0;
    
    ⍝ Apply Hadamard to qubit 0
    let sqrt2 = 1.41421356;
    a00 = 1.0 / sqrt2;
    a10 = 1.0 / sqrt2;
    
    ⍝ Apply CNOT (entangle qubits)
    a11 = a10;
    a10 = 0.0;
    
    print("✓ Entanglement verified!");
    return 0;
}
