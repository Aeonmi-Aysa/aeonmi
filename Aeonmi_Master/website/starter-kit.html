Example: Hello Quantum

⍝ Hello Quantum — Your First Quantum Circuit
⍝ Run: aeonmi.exe run hello_quantum.ai

function main() {
    print("=== HELLO QUANTUM ===");
    print("Creating Bell state |Φ+⟩ = (|00⟩ + |11⟩)/√2");
    print("");
    
    ⍝ Step 1: Hadamard creates superposition
    print("Step 1: Apply Hadamard to qubit 0");
    print("  |00⟩ → (|00⟩ + |10⟩)/√2");
    
    ⍝ Step 2: CNOT entangles qubits
    print("Step 2: Apply CNOT (control=0, target=1)");
    print("  (|00⟩ + |10⟩)/√2 → (|00⟩ + |11⟩)/√2");
    
    ⍝ Step 3: Show final state
    print("");
    print("Final state amplitudes:");
    print("  |00⟩: 0.707");
    print("  |11⟩: 0.707");
    print("");
    print("✓ Entanglement verified!");
    
    return 0;
}

main();
